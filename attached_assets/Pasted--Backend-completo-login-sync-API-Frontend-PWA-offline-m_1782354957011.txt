const express = require("express");
const sqlite3 = require("sqlite3").verbose();
const bcrypt = require("bcryptjs");
const jwt = require("jsonwebtoken");
const multer = require("multer");
const PDFDocument = require("pdfkit");
const turf = require("@turf/turf");
const fs = require("fs");

const app = express();
app.use(express.json());
app.use(express.static("public"));

const SECRET = "incendio_secret_key";

// DB
const db = new sqlite3.Database("./db.sqlite");

// =====================
// INIT DB
// =====================
db.serialize(() => {
  db.run(`CREATE TABLE IF NOT EXISTS users (
    id INTEGER PRIMARY KEY,
    username TEXT,
    password TEXT,
    team TEXT
  )`);

  db.run(`CREATE TABLE IF NOT EXISTS fires (
    id INTEGER PRIMARY KEY,
    data TEXT,
    area REAL,
    team TEXT,
    createdAt TEXT
  )`);
});

// =====================
// LOGIN
// =====================
app.post("/login", (req, res) => {
  const { username, password } = req.body;

  db.get("SELECT * FROM users WHERE username=?", [username], (err, user) => {
    if (!user) return res.json({ error: "user not found" });

    const ok = bcrypt.compareSync(password, user.password);
    if (!ok) return res.json({ error: "invalid" });

    const token = jwt.sign({ id: user.id, team: user.team }, SECRET);
    res.json({ token });
  });
});

// =====================
// SAVE FIRE
// =====================
app.post("/fire", (req, res) => {
  const { token, data, polygon } = req.body;

  const user = jwt.verify(token, SECRET);

  // 🔥 CALCULAR ÁREA (hectares)
  const poly = turf.polygon([polygon]);
  const area = turf.area(poly) / 10000; // m² → ha

  db.run(
    "INSERT INTO fires (data, area, team, createdAt) VALUES (?,?,?,?)",
    [JSON.stringify(data), area, user.team, new Date().toISOString()]
  );

  res.json({ ok: true, area });
});

// =====================
// DASHBOARD
// =====================
app.get("/dashboard", (req, res) => {
  db.all("SELECT * FROM fires", (err, rows) => {
    const total = rows.length;
    const areaTotal = rows.reduce((a, b) => a + b.area, 0);

    res.json({
      total,
      areaTotal,
      rows
    });
  });
});

// =====================
// PDF RELATÓRIO
// =====================
app.get("/report/:id", (req, res) => {
  const id = req.params.id;

  db.get("SELECT * FROM fires WHERE id=?", [id], (err, row) => {
    const doc = new PDFDocument();

    res.setHeader("Content-Type", "application/pdf");
    doc.pipe(res);

    doc.fontSize(20).text("RELATÓRIO DE INCÊNDIO FLORESTAL");
    doc.moveDown();

    doc.fontSize(12).text("Área queimada: " + row.area.toFixed(2) + " ha");
    doc.text("Data: " + row.createdAt);
    doc.text("Equipe: " + row.team);

    doc.end();
  });
});

app.listen(3000, () => console.log("V3 rodando"));

map.addLayer(drawn);

let coords = [];

map.on("draw:created", e => {
  drawn.addLayer(e.layer);
  coords = e.layer.getLatLngs()[0].map(p => [p.lng, p.lat]); // KML order
});

  };

  const res = await fetch("/fire", {

  });

}
